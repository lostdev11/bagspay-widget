This is an example integration of the BagsPay checkout widget.

Note: Make sure the widget server is running at http://localhost:3001 or update the script source below to point to your deployed widget URL.

Checkout

Integration Code:

<script src="http://localhost:3001/api/embed"></script>
<div id="bagspay-checkout"></div>
<script>
  BagsPay.init({
    amount: 29.99,
    merchantId: 'demo-merchant-123',
    orderId: 'order-456',
    onSuccess: (signature) => {
      console.log('Payment successful:', signature);
      alert('Payment successful! Transaction: ' + signature);
    },
    onError: (error) => {
      console.error('Payment failed:', error);
      alert('Payment failed: ' + error.message);
    }
  });
</script>
